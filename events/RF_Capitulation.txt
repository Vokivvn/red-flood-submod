# Hidden event to peace out internationale members
country_event = {
	id = capt.10
	immediate = { log = "[GetDateText]: [Root.GetName]: event capt.10" }

	is_triggered_only = yes
	hidden = yes

	immediate = {
		# First European faction leaders peace out with any non-neighbors outside of Europe
		if = {
			limit = {
				is_in_europe = yes
				is_faction_leader = yes
			}
			every_country = {
				limit = {
					has_country_flag = former_internationale_member
					is_valid_peace_target = yes
					has_war_with = ROOT
					OR = {
						is_in_europe = no
						AND = {
							ROOT = { tag = RUS }
							NOT = {
								is_actual_neighbor_of_ROOT_or_ally = yes
								any_owned_state = {
									any_neighbor_state = {
										owner = {
											is_actual_neighbor_of_ROOT_or_ally = yes
										}
									}
								}
							}
						}	
					}
					OR = {
						tag = BRA
						NOT = {
							tag = GER
							has_capitulated = yes
							is_in_europe = yes
							AND = {
								tag = AUS
								ROOT = {
									OR = {
										tag = SER
										tag = ROM
										tag = POL
										tag = ITA
									}
								}
							}
							is_actual_neighbor_of_ROOT_or_ally = yes
							any_owned_state = {
								is_coastal = yes
								is_controlled_by_ROOT_or_ally = yes
							}
							any_country = {
								is_subject_of = PREV
								OR = {
									is_actual_neighbor_of_ROOT_or_ally = yes
									any_owned_state = {
										is_coastal = yes
										is_controlled_by_ROOT_or_ally = yes
									}
								}
							}
							AND = {
								is_in_faction = yes
								is_faction_leader = yes
								any_country = {
									is_in_faction_with = PREV
									is_faction_leader = no
									OR = {
										is_actual_neighbor_of_ROOT_or_ally = yes
										any_owned_state = {
											is_coastal = yes
											is_controlled_by_ROOT_or_ally = yes
										}
									}
								}
							}
						}
					}
				}
				white_peace_with_ROOT = yes
			}
		}
		# Then Asian faction leaders peace out with any non-neighbours outside of Asia
		else_if = {
			limit = {
				is_in_asia = yes
				is_faction_leader = yes
			}
			every_country = {
				limit = {
					has_country_flag = former_internationale_member
					is_valid_peace_target = yes
					has_war_with = ROOT
					NOT = {
						tag = GER
						has_capitulated = yes
						is_in_asia = yes
						AND = {
							tag = HOL
							ROOT = { tag = DEI }
						}
						is_actual_neighbor_of_ROOT_or_ally = yes
						any_owned_state = {
							is_coastal = yes
							is_controlled_by_ROOT_or_ally = yes
						}
						any_country = {
							is_subject_of = PREV
							OR = {
								is_actual_neighbor_of_ROOT_or_ally = yes
								any_owned_state = {
									is_coastal = yes
									is_controlled_by_ROOT_or_ally = yes
								}
							}
						}
					}
				}
				white_peace_with_ROOT = yes
			}
		}
		# Then everyone else peaces out with any non-neighbours
		else = {
			every_country = {
				limit = {
					has_country_flag = former_internationale_member
					is_valid_peace_target = yes
					has_war_with = ROOT
					OR = {
						tag = BRA
						NOT = {
							tag = GER
							has_capitulated = yes
							AND = {
								tag = HOL
								ROOT = { tag = DEI }
							}
							AND = {
								tag = AUS
								ROOT = {
									OR = {
										tag = SER
										tag = ROM
										tag = POL
										tag = ITA
									}
								}
							}
							is_actual_neighbor_of_ROOT_or_ally = yes
							any_owned_state = {
								is_coastal = yes
								is_controlled_by_ROOT_or_ally = yes
							}
							any_country = {
								is_subject_of = PREV
								OR = {
									is_actual_neighbor_of_ROOT_or_ally = yes
									any_owned_state = {
										is_coastal = yes
										is_controlled_by_ROOT_or_ally = yes
									}
								}
							}
							AND = {
								is_in_faction = yes
